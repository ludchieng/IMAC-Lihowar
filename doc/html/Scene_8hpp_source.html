163  template<typename T>
164  std::list<std::shared_ptr<T>> &findAll() {
165  auto res = new std::list<std::shared_ptr<T>>();
166 
167  // Find all in the collection of islands game objects
168  auto itIslands = _islands.begin();
169  while (itIslands != _islands.end()) {
170  // call findAll on subobjects and concatenate lists
171  res->splice(res->end(), (**itIslands).findAll<T>());
172  ++itIslands;
173  }
174 
175  // Find all in the "other" game objects collection
176  auto itObjects = _objects.begin();
177  while (itObjects != _objects.end()) {
178  // call findAll on subobjects and concatenate lists
179  res->splice(res->end(), (**itObjects).findAll<T>());
180  ++itObjects;
181  }
182  return *res;
183  }
