147  static void load(int argc, char** argv)
148  {
149  EXEC_PATH = argv[0];
150  EXEC_DIR = EXEC_PATH.substr(0, EXEC_PATH.find_last_of("\\/")); // Removes file name
151  std::string defaultConfigFilePath = EXEC_DIR + "/config/default.json";
152  std::string configFilePath = (argc > 1) ? argv[1] : defaultConfigFilePath;
153 
154  try {
155  // First try with user input config file
156  ConfigSerializer::load(configFilePath);
157  loadValues();
158  } catch (LihowarIOException &e) {
159  std::cerr << e.what() << std::endl;
160  std::cerr << "Falling back on default config..." << std::endl;
161 
162  try {
163  // Second try with default input config file
164  ConfigSerializer::load(defaultConfigFilePath);
165  loadValues();
166  } catch (LihowarIOException &e) {
167  // Resigning on hardcoded config
168  std::cerr << e.what() << std::endl;
169  std::cerr << "Falling back on hardcoded config..." << std::endl;
170  }
171 
172  }
